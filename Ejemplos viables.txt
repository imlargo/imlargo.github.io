function BuscarMapDepartamento() {
    let nCasos = 0;
    ///Limpiamos la lista de resultados
    document.getElementById("lstResGis").innerHTML = ""


    ///Verificamos el valor de la capa visible

    //Limpiamos cualquier marca en el mapa
    for (let i = 0; i < MrkAntecedente.length; i++) {

        //Limpia información de mi mapa principal
        map.removeLayer(MrkAntecedente[i]);
        //Limpia informacion de mi mapa de impresión
        //mapPrint.removeLayer(MrkAntecedente2Print[i]);
        //Estas marcas estan en este módulo y script
    }


    //Miramos el valor en la lista departamento
    let vlstDep = document.getElementById('lstDepMn').value;

    for (let i = 0; i < Eventos.length; i++) {

        if (Eventos[i].Departamento == vlstDep) {
            nCasos = nCasos + 1;
            let vDepartamento = Eventos[i].Departamento;
            let vYear = Eventos[i].Year;
            let vTipo = Eventos[i].Tipo;
            let vPunto = Eventos[i].Municipio;
            let vInd = Eventos[i].ind;
            let vLat = Eventos[i].Lat;
            let vLng = Eventos[i].Lng;

            document.getElementById("lbTotalEncontrado").textContent = nCasos;

            let p = document.createElement("p");
            let l = document.createElement("label")
            l.textContent = nCasos + ". Caso: " + vInd + ", " + vYear + ", " + vPunto;
            l.onclick = () => verCaso(i);
            l.classList.add('text-success');
            p.appendChild(l);
            document.getElementById("lstResGis").appendChild(p);

            ///Colocamos las marcas en el mapa
            MrkAntecedente.push(new L.marker([vLat, vLng], { icon: greenIcon })
                .addTo(map)
                .bindPopup("<b>" + vDepartamento + "-" + vYear + "</b><br>" + vPunto + " C:" + vInd + "<br><button type='button' class='btn btn-secondary' onclick ='verCaso(" + i + ")'>Ver</button></br>")
            );


        }

    }

    //Colocamso información de resultados en este espacio

    nCasos = 0;

}

function PutDepLstMap() {
    let DepExis = [];
    for (var i = 1; i < Eventos.length; i++) {
        if (DepExis.includes(Eventos[i].Departamento) == false) {
            DepExis.push(Eventos[i].Departamento);
        }
    }
    //Ordeno los años de menor a mayor
    let DepSort = DepExis.sort();

    let lista = document.getElementById("lstDepMn")
    lista.innerHTML = "";

    for (var i = 0; i < DepSort.length; i++) {
        let nItem = document.createElement("option");
        nItem.value = DepSort[i];
        nItem.text = DepSort[i];
        lista.appendChild(nItem);
    }
}
